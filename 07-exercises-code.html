Python — Exercise 1: Complete Multi-Tool Agent
import anthropic, json

client = anthropic.Anthropic()

# ── Step 1: Define differentiated tools ──
tools = [
    {
        "name": "get_customer",
        "description": """Retrieve verified customer profile by customer ID or email.
Returns: customer_id, name, email, account_status, verification_level, tier.
ALWAYS call this FIRST before any order lookups or refund operations.
Do NOT use for order-specific queries — use lookup_order instead.
Input: customer ID (CUST-XXXXX) or email address.""",
        "input_schema": {
            "type": "object",
            "properties": {"identifier": {"type": "string"}},
            "required": ["identifier"]
        }
    },
    {
        "name": "lookup_order",
        "description": """Retrieve order details by order ID.
Returns: order_id, status, items[], total, created_at, return_eligible.
Requires verified customer_id from get_customer first.
Input: order ID in format ORD-XXXXX only.
Do NOT use customer names or emails — use get_customer for those.""",
        "input_schema": {
            "type": "object",
            "properties": {"order_id": {"type": "string"}},
            "required": ["order_id"]
        }
    },
    {
        "name": "process_refund",
        "description": """Process a refund for a verified customer's order.
Requires: verified customer_id AND order_id from prior tool calls.
Refunds over $500 will be blocked by policy — escalate to human.
Returns: refund_id, status, amount, estimated_arrival.""",
        "input_schema": {
            "type": "object",
            "properties": {
                "customer_id": {"type": "string"},
                "order_id": {"type": "string"},
                "amount": {"type": "number"},
                "reason": {"type": "string"}
            },
            "required": ["customer_id", "order_id", "amount"]
        }
    },
    {
        "name": "escalate_to_human",
        "description": """Escalate to human agent with structured handoff.
Use when: customer explicitly requests human, policy gap exists,
or agent cannot make meaningful progress.""",
        "input_schema": {
            "type": "object",
            "properties": {
                "customer_id": {"type": "string"},
                "summary": {"type": "string"},
                "root_cause": {"type": "string"},
                "recommended_action": {"type": "string"}
            },
            "required": ["customer_id", "summary"]
        }
    }
]

# ── Step 2: Agentic loop with stop_reason ──
system_prompt = """You are a customer support agent. Follow these rules:
1. Always verify customer with get_customer BEFORE any order operations
2. Decompose multi-issue messages into separate concerns
3. Escalate immediately if: customer requests human, policy gap, no progress
4. For straightforward issues (standard returns with evidence), resolve autonomously
5. For complex issues (policy exceptions), escalate with structured handoff"""

def run_agent(user_message):
    messages = [{"role": "user", "content": user_message}]
    verified_customer = None

    while True:
        response = client.messages.create(
            model="claude-sonnet-4-20250514",
            max_tokens=4096,
            system=system_prompt,
            tools=tools,
            messages=messages
        )

        # ✅ stop_reason drives the loop
        if response.stop_reason == "end_turn":
            return response.content

        if response.stop_reason == "tool_use":
            messages.append({"role": "assistant", "content": response.content})

            tool_results = []
            for block in response.content:
                if block.type == "tool_use":
                    # ── Step 4: Policy hook ──
                    if block.name == "process_refund":
                        amount = block.input.get("amount", 0)
                        if amount > 500:
                            tool_results.append({
                                "type": "tool_result",
                                "tool_use_id": block.id,
                                "content": json.dumps({
                                    "blocked": True,
                                    "reason": f"Refund ${amount} exceeds $500 threshold",
                                    "action": "Please escalate to human agent"
                                })
                            })
                            continue

                    # Execute tool and get result
                    result = execute_tool(block.name, block.input)
                    tool_results.append({
                        "type": "tool_result",
                        "tool_use_id": block.id,
                        "content": json.dumps(result)
                    })

            messages.append({"role": "user", "content": tool_results})
